Code void crossNumAndVectorToOut(num s, Vector a, Vector out) #

Has the effect of swapping the x and y coordinates of the vector, multiplying both by the given number, and then flipping the sign of the new x coordinate. Returns the result through the out parameter.

static void crossNumAndVectorToOut(num s, Vector a, Vector out) {
  num tempY = s * a.x;
  out.x = -s * a.y;
  out.y = tempY;
}

Code void maxToOut(Vector a, Vector b, Vector out) #

Take the maximum of each coordinate from the two given vectors and store the result in the given out vector.

static void maxToOut(Vector a, Vector b, Vector out) {
  out.x = a.x > b.x ? a.x : b.x;
  out.y = a.y > b.y ? a.y : b.y;
}

Code void minToOut(Vector a, Vector b, Vector out) #

Take the minimum of each coordinate from the two given vectors and store the result in the given out vector.

static void minToOut(Vector a, Vector b, Vector out) {
  out.x = a.x < b.x ? a.x : b.x;
  out.y = a.y < b.y ? a.y : b.y;
}

Code num distanceBetween(Vector v) #

Returns the distance between the point defined by this vector and the given vector.

num distanceBetween(Vector v) {
  num xDelta = this.x - v.x;
  num yDelta = this.y - v.y;
  return Math.sqrt(xDelta * xDelta + yDelta * yDelta);
}

Code num normalize() #

Normalizes this vector and returns the length before normalization.

num normalize() {
  num len = length;
  if (len < Settings.EPSILON) {
    return 0;
  }

  num invLength = 1.0 / len;
  x *= invLength;
  y *= invLength;
  return len;
}

Code Vector rotate(num angle) #

Returns a vector that is the result of rotating the original vector by angle degrees counter-clockwise.

Vector rotate(num angle) {
  Matrix22 mtx = new Matrix22();
  mtx.setAngle(angle);
  Vector temp = new Vector();
  mtx.multiplyVectorToOut(this, temp);
  return temp;
}
