class WorldQueryWrapper implements TreeCallback {
  BroadPhase broadPhase;
  QueryCallback callback;

  WorldQueryWrapper() { }

  bool treeCallback(DynamicTreeNode node) {
    Fixture fixture = node.userData;
    return callback.reportFixture(fixture);
